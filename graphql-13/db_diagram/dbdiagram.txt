// https://dbdiagram.io/d

Table users {
  id uuid [pk]
  name varchar
  email varchar
  password varchar
  registeredAt timestamp
}

Table orders {
  id uuid [pk]
  date timestamp
  status varchar
  userId uuid
}

Table ordered_products {
  orderId uuid [pk]
  productId uuid [pk]
  name varchar
  price money
  quantity int
}

Table categories {
  id uuid [pk]
  name varchar
}

Table products {
  id uuid [pk]
  name varchar
  price money
  categoryId uuid
}
